# 0.3 — Define All Core Types & Initial Commit

```
In the "cowork-orchestrator" project, implement `src/core/types.ts` with ALL foundational interfaces. This is the single source of truth for types across the entire project. Export everything.

Define these types:

// Task status lifecycle
type TaskStatus = 'pending' | 'queued' | 'assigned' | 'running' | 'paused' | 'completed' | 'failed' | 'cancelled';

// Task priority
type TaskPriority = 'critical' | 'high' | 'medium' | 'low';

// Agent capability descriptor
interface AgentCapability {
  name: string;
  description: string;
  inputSchema: z.ZodSchema;
  outputSchema: z.ZodSchema;
  costEstimate?: 'low' | 'medium' | 'high';
}

// Task definition
interface Task {
  id: string;
  parentId?: string;
  name: string;
  description: string;
  priority: TaskPriority;
  status: TaskStatus;
  assignedAgentId?: string;
  input: Record<string, unknown>;
  output?: Record<string, unknown>;
  dependencies: string[];
  subtasks: string[];
  metadata: TaskMetadata;
  createdAt: Date;
  updatedAt: Date;
  timeout?: number;
}

interface TaskMetadata {
  estimatedTokens?: number;
  actualTokens?: number;
  retryCount: number;
  maxRetries: number;
  checkpointId?: string;
}

// TaskResult — returned by agents after execution
interface TaskResult {
  taskId: string;
  success: boolean;
  output: Record<string, unknown>;
  error?: string;
  tokenUsage?: { input: number; output: number };
  duration: number;
}

// Agent definition
interface AgentConfig {
  id: string;
  name: string;
  role: string;
  systemPrompt: string;
  capabilities: AgentCapability[];
  maxConcurrentTasks: number;
  model: string;
  temperature?: number;
  maxTokens?: number;
  tools?: ToolDefinition[];
}

// Agent status
type AgentStatus = 'idle' | 'busy' | 'error' | 'offline';

// Agent selector config (used in workflow nodes)
interface AgentSelectorConfig {
  strategy: 'capability-match' | 'least-loaded' | 'round-robin' | 'ai-selected';
  preferredAgentId?: string;
  requiredCapabilities?: string[];
}

// Workflow definition (DAG)
interface Workflow {
  id: string;
  name: string;
  description: string;
  nodes: WorkflowNode[];
  edges: WorkflowEdge[];
  status: TaskStatus;
  context: Record<string, unknown>;
}

interface WorkflowNode {
  id: string;
  taskTemplate: Omit<Task, 'id' | 'status' | 'createdAt' | 'updatedAt'>;
  agentSelector?: AgentSelectorConfig;
}

interface WorkflowEdge {
  from: string;
  to: string;
  condition?: string;
}

interface WorkflowResult {
  workflowId: string;
  success: boolean;
  outputs: Map<string, TaskResult>;
  duration: number;
  nodesCompleted: number;
  nodesTotal: number;
}

// Message types for inter-agent communication
interface AgentMessage {
  id: string;
  from: string;
  to: string | '*';
  type: 'request' | 'response' | 'event' | 'delegation' | 'negotiation';
  channel: string;
  payload: unknown;
  timestamp: Date;
  correlationId?: string;
}

// Orchestration config
interface OrchestratorConfig {
  maxConcurrentAgents: number;
  maxConcurrentTasks: number;
  defaultTimeout: number;
  checkpointInterval: number;
  rateLimits: {
    requestsPerMinute: number;
    tokensPerMinute: number;
  };
  persistence: {
    enabled: boolean;
    dbPath: string;
  };
}

// MCP Tool definition
interface ToolDefinition {
  name: string;
  description: string;
  inputSchema: Record<string, unknown>;
  handler?: string;
}

// Tool execution result
interface ToolResult {
  success: boolean;
  output: unknown;
  error?: string;
  duration: number;
}

// Tool call (from Claude API response)
interface ToolCall {
  toolName: string;
  input: unknown;
  id: string;
}

// Audit log entry
interface AuditEntry {
  eventType: string;
  agentId?: string;
  taskId?: string;
  workflowId?: string;
  data: Record<string, unknown>;
  timestamp: Date;
}

After writing types.ts, run `npx tsc --noEmit` to verify it compiles. Fix any issues.

Then make the initial git commit: "chore: project scaffold and type definitions"
```
