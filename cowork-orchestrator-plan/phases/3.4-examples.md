## 2. examples/simple-task.ts — Minimal Single-Task Example

Create a minimal working example:
```typescript
// Simple example: One agent, one task
// Usage: GEMINI_API_KEY=your-key npx ts-node examples/simple-task.ts

import { OrchestrationEngine } from '../src';
import dotenv from 'dotenv';
dotenv.config();

async function main() {
  if (!process.env.GEMINI_API_KEY) {
    console.error('Please set GEMINI_API_KEY in your .env file');
    process.exit(1);
  }

  const engine = new OrchestrationEngine();
  await engine.start();

  // Register a general-purpose agent
  engine.registerAgent({
    id: 'general-1',
    name: 'General Assistant',
    role: 'general',
    systemPrompt: 'You are a helpful assistant. Complete tasks concisely and accurately.',
    capabilities: [{ name: 'general', description: 'General purpose tasks', inputSchema: {}, outputSchema: {} }],
    maxConcurrentTasks: 3,
    model: 'gemini-2.0-flash'
  });

  const result = await engine.submitTask({
    name: 'summarize-typescript',
    description: 'Summarize the key benefits of TypeScript over JavaScript in 3 bullet points',
    priority: 'medium',
    input: {}
  });

  console.log('Result:', JSON.stringify(result, null, 2));
  console.log('\nDashboard:\n', engine.getDashboard());

  await engine.stop();
}

main().catch(console.error);
```
